rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // Users collection rules
    match /users/{userId} {
      // Allow read for all authenticated users (needed for profiles)
      allow read: if true;
      
      allow create: if request.auth != null
        && request.auth.uid == userId
        && request.resource.data.keys().hasAll(['first_name', 'last_name', 'email', 'username', 'user_id'])
        && request.resource.data.first_name is string
        && request.resource.data.last_name is string
        && request.resource.data.email is string
        && request.resource.data.username is string
        && request.resource.data.user_id is string
        && request.resource.data.user_id == userId
        && request.resource.data.email == request.auth.token.email;

      allow update, delete: if request.auth != null
        && request.auth.uid == userId;
    }

    // Listings collection - PERMISSIVE for marketplace
    match /tbl_listing/{listingId} {
      // Anyone can read listings
      allow read: if true;
      
      // Anyone can create listings
      allow create: if 
        request.resource.data.title is string
        && request.resource.data.description is string
        && request.resource.data.price is number
        && request.resource.data.category is string
        && request.resource.data.user_id is string;
      
      // Users can update/delete their own listings
      allow update, delete: if true;
    }

    // ========== SAVED PRODUCTS COLLECTION ==========
    // Each user can save products - stored with user_id
    match /tbl_saved/{savedId} {
      // Anyone can read (for now, to simplify)
      allow read: if true;
      
      // Anyone can create saved items
      allow create: if 
        request.resource.data.user_id is string
        && request.resource.data.product_id is string;
      
      // Allow update and delete
      allow update, delete: if true;
    }

    // ========== CONVERSATIONS COLLECTION ==========
    match /conversations/{conversationId} {
      allow read, write: if true;
    }

    // ========== MESSAGES SUBCOLLECTION ==========
    match /conversations/{conversationId}/messages/{messageId} {
      allow read, write: if true;
    }

    // ========== GEO COLLECTIONS (Read-only) ==========
    match /ph_regions/{docId} {
      allow read: if true;
    }
    match /ph_provinces/{docId} {
      allow read: if true;
    }
    match /ph_cities/{docId} {
      allow read: if true;
    }
    match /ph_barangays/{docId} {
      allow read: if true;
    }

    // Deny all other paths by default
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
