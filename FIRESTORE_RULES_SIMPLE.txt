rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // Allow all reads
    match /{document=**} {
      allow read: if true;
    }

    // Users collection: only authenticated users can write their own
    match /users/{userId} {
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    // Listings collection: allow anyone to create and update by user_id
    match /tbl_listing/{listingId} {
      allow create: if request.resource.data.keys().hasAll(['title', 'description', 'price', 'category', 'user_id']);
      
      allow update, delete: if resource.data.user_id == request.resource.data.user_id;
    }
  }
}
